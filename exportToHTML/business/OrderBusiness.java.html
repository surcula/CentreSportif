package business;

import Tools.Result;
import dto.OrderActionForm;
import dto.OrderDiscountForm;
import entities.Discount;
import entities.OrdersDiscount;
import org.apache.log4j.Logger;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * Validation de forme + helpers métier pour Order,
 * même syntaxe et façon de coder que SportBusiness.
 */
public class OrderBusiness {

    // Log4j
    private static final Logger log = Logger.getLogger(OrderBusiness.class);

    /* ========================================================= 
       1) INIT FORMS (validation des formulaires POST) 
       ========================================================= */

    /**
     * Valide le formulaire "appliquer une réduction" (orderId, discountId)
     */
    public static Result<OrderDiscountForm> initApplyDiscountForm(String strOrderId, String strDiscountId) {

        Map<String, String> errors = new HashMap<String, String>();

        // Required
        Result<String> orderIdStr = ValidateForm.stringIsEmpty(strOrderId, "errorOrderId", "Order ID", errors);
        if (!orderIdStr.isSuccess()) errors.putAll(orderIdStr.getErrors());

        Result<String> discountIdStr = ValidateForm.stringIsEmpty(strDiscountId, "errorDiscountId", "Discount ID", errors);
        if (!discountIdStr.isSuccess()) errors.putAll(discountIdStr.getErrors());

        // Parse integer
        Result<Integer> orderId = ValidateForm.parseInteger(orderIdStr.getData(), "errorOrderId", "Order ID", errors);
        if (!orderId.isSuccess()) errors.putAll(orderId.getErrors());
        else if (orderId.getData().intValue() <= 0) errors.put("errorOrderId", "Order ID doit être un entier positif.");

        Result<Integer> discountId = ValidateForm.parseInteger(discountIdStr.getData(), "errorDiscountId", "Discount ID", errors);
        if (!discountId.isSuccess()) errors.putAll(discountId.getErrors());
        else if (discountId.getData().intValue() <= 0) errors.put("errorDiscountId", "Discount ID doit être un entier positif.");

        if (!errors.isEmpty()) {
            log.error("initApplyDiscountForm: " + errors);
            return Result.fail(errors);
        }

        // to DTO
        Result<OrderDiscountForm> form = toOrderDiscountForm(orderId.getData(), discountId.getData());
        return Result.ok(form.getData());
    }

    /**
     * Valide un formulaire qui ne contient qu'un orderId (payer via abo / payer en ligne / annuler)
     */
    public static Result<OrderActionForm> initOrderActionForm(String strOrderId) {

        Map<String, String> errors = new HashMap<String, String>();

        // Required
        Result<String> orderIdStr = ValidateForm.stringIsEmpty(strOrderId, "errorOrderId", "Order ID", errors);
        if (!orderIdStr.isSuccess()) errors.putAll(orderIdStr.getErrors());

        // Parse integer
        Result<Integer> orderId = ValidateForm.parseInteger(orderIdStr.getData(), "errorOrderId", "Order ID", errors);
        if (!orderId.isSuccess()) errors.putAll(orderId.getErrors());
        else if (orderId.getData().intValue() <= 0) errors.put("errorOrderId", "Order ID doit être un entier positif.");

        if (!errors.isEmpty()) {
            log.error("initOrderActionForm: " + errors);
            return Result.fail(errors);
        }

        // to DTO
        Result<OrderActionForm> form = toOrderActionForm(orderId.getData());
        return Result.ok(form.getData());
    }

    /* ========================================================= 
       2) CALCUL METIER (application des réductions) 
       ========================================================= */

    /**
     * Applique les remises séquentiellement (percent en 0..100) et borne à 0.
     * Même responsabilité que l’ex OrderBusiness.applyDiscounts(...),
     * mais conservée ici pour coller à ton organisation "Business".
     */
    public static Result<Double> computeDiscountedTotal(double base, List<OrdersDiscount> orderDiscounts) {
        double result = base;
        int count = (orderDiscounts == null) ? 0 : orderDiscounts.size();

        log.debug("computeDiscountedTotal | base=" + base + " | nbDiscounts=" + count);

        if (orderDiscounts != null) {
            for (int i = 0; i < orderDiscounts.size(); i++) {
                OrdersDiscount od = orderDiscounts.get(i);
                Discount d = od.getDiscount();
                if (d != null && d.isActive()) {
                    double pct = d.getPercent(); // 10.0 => 10%
                    result = result * (1.0 - pct / 100.0);
                }
            }
        }
        if (result < 0.0) result = 0.0;

        log.info("computeDiscountedTotal result=" + result);
        return Result.ok(Double.valueOf(result));
    }

    /* ========================================================= 
       3) MAPPERS (comme toEntity/toUpdateForm dans SportBusiness) 
       ========================================================= */

    private static Result<OrderDiscountForm> toOrderDiscountForm(int orderId, int discountId) {
        OrderDiscountForm f = new OrderDiscountForm();
        f.setOrderId(orderId);
        f.setDiscountId(discountId);
        return Result.ok(f);
    }

    private static Result<OrderActionForm> toOrderActionForm(int orderId) {
        OrderActionForm f = new OrderActionForm();
        f.setOrderId(orderId);
        return Result.ok(f);
    }
}
